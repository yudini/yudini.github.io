## Stack
:  LIFO(Last In First Out)구조
### ***쌓아올리는 개념. 제일 위에(늦게)쌓인 데이터가 제일 먼저 빠져나온다.***

 제일 위에 있는 것을 먼저 빼는 느낌 

<br>
<br>
linked list를 사용해서 스택 구현

~~~c++
#include <stdio.h>
#include <stdlib.h>


struct node{
    int data;
    struct node *next;
};

typedef struct node node;

node *temp, *head, *tail;
void insert(int);
void pop();
void print();
void release();

int main(){
    int n,x;
    head=malloc(sizeof(struct node));
    tail=malloc(sizeof(struct node));
    head->next=tail;

    while(1){
        printf("1.삽입 2.값 꺼내기 3.출력 0.끝 \n");
        scanf("%d",&n);
        switch(n){
        case 1:
            printf("값 입력: ");
            scanf("%d",&x);
            insert(x);
            break;
        case 2:
            pop();
            break;
        case 3:
            print();
            break;
        case 0:
            release();
            break;
        }
    }

}

void insert(int x){
    temp=malloc(sizeof(struct node));
    temp->data=x;
    temp ->next=head->next;
    head->next = temp;
}

void pop(){
    if(head->next==tail){
        printf("Stack is empty! \n");
        return;
    }
    temp=head->next;
    head->next=temp->next;
    printf("%d pop \n",temp->data);
    free(temp);
}

void release(){
    temp=head->next;
    while(temp!=tail){
        head->next=temp->next;
        free(temp);
    }
    free(head);
    free(tail);
}
void print(){
    temp=head->next;
    if(temp==tail){
        printf("Stack is empty! \n");
        return;
    }
    while(temp!=tail){
        printf("%d ",temp->data);
        temp=temp->next;                                                                 
    }
    printf("\n");
}
 



~~~
